typescript
import { NgModule } from '@angular/core';

import { ReactiveFormsModule } from '@angular/forms'; // Importa ReactiveFormsModule

import { IonicModule } from '@ionic/angular';

import { PeliculasadminPageRoutingModule } from './peliculasadmin-routing.module';

Asegúrate de agregar ReactiveFormsModule a la lista de imports. Después de hacer esto, Angular debería reconocer la directiva formGroup en tu formulario.
